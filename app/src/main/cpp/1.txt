char *
strcat(char *s, const char *append)
{
	char *save = s;
                                       	for (; *s; ++s);
                                       	while ((*s++ = *append++) != '\0');
	return(save);
}

char *
strcpy(char *to, const char *from)
{
	char *save = to;
	for (; (*to = *from) != '\0'; ++from, ++to);
	return(save);
}
